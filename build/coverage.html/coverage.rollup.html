    var args
      , arg
      , cmds
      , callback
      , opts = {
          interactive: false

        , breakOnError: true

      cmds = [cmds];
    }
    // Make a copy if it's an actual list
    else {
      cmds = cmds.slice();
    }

    while((arg = args.shift())) {
      if (typeof arg == 'function') {
        callback = arg;

      delete opts.stderr;
    }
